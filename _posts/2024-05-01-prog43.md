# __part43. 요격 시스템__

### _js ver 1.0_
```js 
function solution(targets){
    // [[11, 13], [10, 14], [5, 12], [4, 5], [4, 8], [3, 7], [1, 4]]
    // prv: 최대치    missile: 0
    // prv: 11        missile: 1
    // prv: 11        missile: 1
    // prv: 11        missile: 1
    // prv: 4         missile: 2
    // prv: 4         missile: 2
    // prv: 4         missile: 2
    // prv: 1         missile: 3

    targets.sort((a,b) => b[0] - a[0]);

    let prv = Infinity;

    return targets.reduce((missile, [start, end]) => {
        
        if (end > prv || start >= prv) {
            prv = Math.max(prv, start);
        } else {
            prv = start;
            missile++;
        }

        return missile;
    }, 0)
}
```
